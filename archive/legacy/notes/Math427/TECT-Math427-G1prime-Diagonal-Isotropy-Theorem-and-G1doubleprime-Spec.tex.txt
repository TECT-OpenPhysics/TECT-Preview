\subsection*{1. Proposition and proof}

\paragraph{Setting.} Gibbs--Bogoliubov variational free energy at
condensate amplitude $A$ for reading $\mathcal R$ ($n$ pairs, exact
coefficients $N_4, N_6$ per Math424-AddA \S 2), with diagonal trial
propagator $G^{-1}(q) = \hat r(\hat q) + c\,(q^2 - q_0^2)^2$:
\[
F_{\rm var}[\hat r(\cdot), A]/V
= \tfrac12 \!\int_q \ln G^{-1}(q)
- \tfrac12 \!\int_q \hat r(\hat q)\, G(q)
+ \Phi\big(M_{\rm tot}, A\big),
\qquad
M_{\rm tot} = \int_q G(q),
\]
where $\Phi(M, A)$ collects the condensate terms and ALL interaction
terms; because the quartic/sextic interactions are local and the
Gaussian variance $\langle \delta\phi^2(x) \rangle = M_{\rm tot}$ is
position-independent for any homogeneous Gaussian state, $\Phi$ depends
on the trial profile ONLY through the scalar $M_{\rm tot}$ (the $c$-term
contributions cancel between $\langle H \rangle$ and
$\langle H_{\rm trial} \rangle$ exactly as in the isotropic case).

\paragraph{Proof.} Vary $\hat r(\hat q)$ pointwise in direction $\hat q$.
Writing the radial loop integrals $M_{\rm rad}(\rho)$ and using
$\partial M_{\rm tot} / \partial \hat r(\hat q) =
(d\Omega/4\pi)\, M_{\rm rad}'(\hat r(\hat q))$,
\[
\frac{\delta F_{\rm var}}{\delta \hat r(\hat q)}
= \frac{d\Omega}{4\pi}\, M_{\rm rad}'(\hat r(\hat q))
\left[ -\tfrac12 \hat r(\hat q) + \Phi'(M_{\rm tot}, A) \right]
\;=\; 0
\quad\Longrightarrow\quad
\hat r(\hat q) = 2\,\Phi'(M_{\rm tot}, A).
\]
The right-hand side is direction-independent, so every stationary
profile is constant in $\hat q$ (isotropic), and the stationary value
reproduces exactly the isotropic ordered-phase gap equation of
Math424-AddA (G). Boundary escape is excluded: as $\hat r \to 0^+$ on
any direction set of positive measure, $M_{\rm rad} \to \infty$ (the
on-shell integrand $\propto 1/(\hat r + c(q^2-q_0^2)^2)$ diverges) and
the sextic term $\tfrac{5v}{2} M_{\rm tot}^3 \to +\infty$ dominates, so
$F_{\rm var} \to +\infty$; as $\hat r \to \infty$, $F_{\rm var}$ grows
through $-\tfrac12\hat r M + \tfrac12 \ln$-term balance. Hence the
infimum is attained at the (unique, per the Math400-AddE-type gap-map
monotonicity) isotropic stationary point. $\square$
