\begin{abstract}
The Newton-Krylov v2.3 proof protocol for the TECT first-pillar claim
$m^{*2}>0$ currently accepts, as the physical spectral gap, the smallest
strictly positive Ritz eigenvalue of the projected lattice Hessian.  The
recent $N{=}32 \to N{=}64$ continuum audit produced
$m^{*2}(N{=}32)=3.1485 \to m^{*2}(N{=}64)=54.07$, a factor-of-17 jump
inconsistent with the continuum expansion
$m^{*2}(h^2) = m^{*2}_0 + c\,h^2 + O(h^4)$ with $h(N)=2\pi L/N$.  We
establish two complementary layers of analysis.
\textbf{Theoretically}, we prove that the projected Hessian admits a
direct-sum decomposition over three wavenumber-disjoint invariant
subspaces -- infrared (IR), first-shell (shell), and ultraviolet grid
(UV) -- and that only the IR block carries grid-invariant eigenvalues
(Theorems~\ref{thm:decomp}--\ref{thm:sound}).  This provides the
definition of a grid-invariant Phase-2.5 acceptance gate.
\textbf{Empirically}, the companion script
\texttt{PDE/hess\_jump\_audit.py} measures the gate observables on the
saved outputs and refutes the UV-ghost hypothesis: both Ritz pairs are
IR-localised.  The measurement exposes a deeper failure mode --
\emph{Newton-Krylov collapsed to the trivial vacuum on both grids}
($\|\Psi^*\|_{\text{RMS}}/\phi_0 \sim 10^{-6}$), and the anomalous
eigenvalues originate from a class-II backend singularity at the
regularised quotient $m_\alpha/(\rho + 10^{-12})$ when $\rho$ is
comparable to $10^{-12}$.  We retract both $N{=}32$ and $N{=}64$
Phase-2 numbers, augment the Phase-2.5 gate by a Phase-0 vacuum-escape
prerequisite $\|\Psi^*\|_{\text{RMS}} / \phi_0 \ge 0.3$, specify the
required class-II regularisation fix, and redirect Pillar 1 closure
through the Math55 continuation method as the exclusive entry point
to valid BCC Phase-2 runs.
\end{abstract}

\section{Setting and Notation}

Let $\Lambda_N = a\mathbb{Z}_N^3$ be the periodic cubic lattice of side
$L = a N$, $a = L/N$, and let $\Psi : \Lambda_N \to \mathbb{C}^3$ be the
three-component order parameter on which the TECT Brazovskii backend
\texttt{hessian\_vec} acts.  Write $\mathcal{H}_N = \mathbb{C}^{3 N^3}$
for the complex configuration space and $\mathcal{H}_N^{\mathbb{R}}$ for
its real-flattened counterpart of dimension $\dim \mathcal{H}_N^{\mathbb{R}}
= 6 N^3$.  Fourier modes are labelled
$\mathbf{k} = (2\pi/L)\mathbf{n}$ with
$\mathbf{n} \in \mathbb{Z}_N^3 \cap (-N/2, N/2]^3$; the Nyquist
wavenumber is $k_{\text{Ny}} = \pi/a = \pi N/L$.  Define the Brazovskii
dispersion
\begin{equation}
  \omega(k) \;=\; r + Z k^2 + Y k^4,
  \qquad
  q_0 \;=\; \sqrt{-Z/(2Y)},
  \qquad
  \omega(q_0) = r - Z^2/(4Y) = r + Y q_0^4,
  \label{eq:omega}
\end{equation}
and the first shell
$\mathcal{S}_1 = \{\mathbf{k} \in \Lambda_N^* : | |\mathbf{k}| - q_0 | < \delta\}$
with $\delta \ll q_0$ fixed (below we take $\delta = q_0/4$).

At a converged BCC condensate $\Psi^{*}$ with $\|R_{\text{proj}}(\Psi^*)\|
< \text{tol}$, the projected Hessian is
\begin{equation}
  H_{\text{proj}} \;\equiv\; P\, H[\Psi^*]\, P,
  \qquad
  P = I - \sum_{\alpha=1}^{n_{\text{zm}}} |e_\alpha\rangle\langle e_\alpha|,
  \label{eq:Hproj}
\end{equation}
where $\{e_\alpha\}$ is the orthonormal zero-mode basis produced by
\texttt{build\_zero\_mode\_projector}.  The v2.3 protocol reports
\begin{equation}
  m^{*2}(N) \;\equiv\; \min\{\lambda \in \sigma(H_{\text{proj}}) : \lambda > \texttt{zero\_tol}\},
  \label{eq:mstar}
\end{equation}
as the physical spectral gap.

\section{Wavenumber-Stratified Decomposition (Theorem 1)}

\begin{definition}[Wavenumber band projectors]\label{def:bands}
Fix cutoffs $0 < k_{\text{IR}} < k_{\text{Ny}}/2$ and $\delta \in (0, k_{\text{IR}}/2)$.
Define three mutually orthogonal Fourier-band projectors on
$\mathcal{H}_N^{\mathbb{R}}$:
\begin{align}
  \Pi_{\text{IR}}   &: \text{support on } |\mathbf{k}| \le 2 q_0, \\
  \Pi_{\text{shell}}&: \text{support on } 2 q_0 < |\mathbf{k}| \le k_{\text{IR}}, \\
  \Pi_{\text{UV}}   &: \text{support on } k_{\text{IR}} < |\mathbf{k}| \le k_{\text{Ny}},
\end{align}
with $\Pi_{\text{IR}}+\Pi_{\text{shell}}+\Pi_{\text{UV}} = I$.  In
practice we take $k_{\text{IR}} = 3 q_0$; Theorem~\ref{thm:decomp} is
insensitive to this choice up to $O(a^2)$ mixing terms.
\end{definition}

\begin{theorem}[Band-diagonal decomposition of $H_{\text{proj}}$]\label{thm:decomp}
Let the BCC condensate $\Psi^*$ have Fourier support essentially confined
to the first shell, $\|(1-\Pi_{\mathcal{S}_1})\Psi^*\|^2/\|\Psi^*\|^2 = \epsilon_{\text{BCC}}$
with $\epsilon_{\text{BCC}} \ll 1$.  Then
\begin{equation}
  H_{\text{proj}}
  \;=\; H_{\text{IR}}^{\text{proj}} \oplus H_{\text{shell}}^{\text{proj}} \oplus H_{\text{UV}}^{\text{proj}}
        + \mathcal{E}(\epsilon_{\text{BCC}}, a^2),
  \label{eq:decomp}
\end{equation}
with block Hamiltonians
$H_{X}^{\text{proj}} = \Pi_X \, H_{\text{proj}} \, \Pi_X$ for
$X \in \{\text{IR}, \text{shell}, \text{UV}\}$, and off-block coupling
\begin{equation}
  \|\Pi_X H_{\text{proj}} \Pi_Y\|
  \;\le\; C_1\, \epsilon_{\text{BCC}}\,(|\lambda| + |\gamma|\,\|\Psi^*\|_\infty^2)
        + C_2\, a^2,
  \qquad X \ne Y,
  \label{eq:offblock}
\end{equation}
for a constant $C_1, C_2$ depending only on $(r, Z, Y)$ and $L$, and
all $\mathbf{k}$ sums restricted to $\Lambda_N^*$.
\end{theorem}

\begin{proof}
The Hessian expands as
\begin{equation}
  H[\Psi^*] v \;=\; H_0 v \;+\; V_{\text{nl}}[\Psi^*] v,
  \qquad
  H_0 v \;=\; r v + Z \Delta v + Y \Delta^2 v,
  \label{eq:Hsplit}
\end{equation}
where $H_0$ is the linear Brazovskii operator (diagonal in Fourier
space with symbol $\omega(k)$ of Eq.~\ref{eq:omega}) and $V_{\text{nl}}$
is the pointwise cubic-quartic multiplier
\begin{equation}
  V_{\text{nl}}[\Psi^*] v
  \;=\; 2 \lambda |\Psi^*|^2 v + \lambda (\Psi^*)^2 v^*
        + 3\gamma|\Psi^*|^4 v + 2\gamma(\Psi^*)^2|\Psi^*|^2 v^*.
  \label{eq:Vnl}
\end{equation}
In Fourier space $V_{\text{nl}}$ is a convolution operator with kernel
supported on the self-convolutions of the BCC support set $\mathcal{S}_1$.
Convolution of $\mathcal{S}_1$ with itself $p$ times produces support in
$\{|\mathbf{k}| \le p\, q_0 \sqrt{3}\}$ (the BCC first shell has
$q_0|(\pm1,\pm1,0)/\sqrt{2}| = q_0$ and its harmonics $2q_0, 3q_0, \ldots$).
Specifically, the kernel of $|\Psi^*|^2 = \Psi^* \overline{\Psi^*}$ is
supported in the second shell $|\mathbf{k}| \le 2 q_0$, and the kernel of
$|\Psi^*|^4$ in $|\mathbf{k}| \le 4 q_0$.  Consequently, for any two
bands $X \ne Y$ with $\mathrm{dist}(X,Y) > 4 q_0$ in wavenumber, the
matrix element $\Pi_X V_{\text{nl}} \Pi_Y$ vanishes exactly.  For
$X,Y$ adjacent, the matrix element is bounded by
$\|V_{\text{nl}}\|_{\ell^1_{\mathbf{k}}} \le \|(2\lambda + 3\gamma\|\Psi^*\|_\infty^2)\Psi^*\|_{L^2}^2
\epsilon_{\text{BCC}}$ (Hausdorff--Young).  $H_0$ is diagonal in Fourier
space and preserves bands exactly on the continuum; the lattice
finite-difference version of $\Delta$ introduces an $O(a^2)$ mixing
through the replacement $k \to k_{\text{lat}}(k) = k + O(a^2 k^3)$.
Combining the two bounds gives Eq.~\eqref{eq:offblock}.
\end{proof}

\begin{remark}[Isospectral content of the block decomposition]\label{rem:content}
To leading order in $(\epsilon_{\text{BCC}}, a^2)$:

\begin{itemize}
\item $H_{\text{IR}}^{\text{proj}}$ acts on the space of long-wavelength
  perturbations $|\mathbf{k}| \le 2 q_0$.  This space contains:
  (i) the Goldstone-like phase modes of the condensate (projected out
  by $P$ via translations), (ii) the amplitude mode
  $\phi \leftrightarrow \phi + \eta$ with $\hat\eta$ localised at the
  first shell (residing partly in the shell band, partly in the IR
  band at $|\mathbf{k}| \to 0$), and (iii) acoustic hydrodynamic modes
  at $|\mathbf{k}| \to 0$.  The \emph{physical spectral gap} is
  \begin{equation}
    m^{*2}_{\text{phys}} \;=\; \min \sigma(H_{\text{IR}}^{\text{proj}}) \setminus\{0\}
    \;=\; O(\mu^2) + O(\lambda,\gamma,\phi_0^2).
  \end{equation}

\item $H_{\text{shell}}^{\text{proj}}$ acts on first-shell orthogonal
  fluctuations (in shell space, orthogonal to the BCC condensate
  direction).  Its eigenvalues are of order $|\mu^2|$ and are
  grid-invariant modulo $O(a^2)$ shell-structure corrections.

\item $H_{\text{UV}}^{\text{proj}}$ acts on modes with
  $|\mathbf{k}| > 3 q_0$.  $V_{\text{nl}}$ is \emph{exponentially
  suppressed} on this block (no BCC support at such $\mathbf{k}$) and
  the spectrum is dominated by $H_0$:
  \begin{equation}
    \sigma(H_{\text{UV}}^{\text{proj}})
    \;=\; \{\omega(k_{\text{lat}}) + O(\lambda) : \mathbf{k} \in \Pi_{\text{UV}}\},
  \end{equation}
  with minimum $\omega(3 q_0) = r + 9 Z q_0^2 + 81 Y q_0^4$ and
  maximum $\omega(k_{\text{Ny}})$.
\end{itemize}
\end{remark}

\section{UV-Scaling Law and the $\times 16$ Prediction (Proposition 3)}

\begin{proposition}[UV block scales as $N^4$]\label{prop:Nfour}
For $Y > 0$ fixed and $N \to \infty$,
\begin{equation}
  \lambda_{\max}\bigl(H_{\text{UV}}^{\text{proj}}(N)\bigr)
  \;=\; Y \, k_{\text{Ny}}^4 (1 + O(a^2))
  \;=\; Y \left(\frac{\pi N}{L}\right)^4 + O(N^2),
  \label{eq:UVscale}
\end{equation}
and the density of UV eigenvalues near $\lambda_{\max}$ grows as $N^3$.
Consequently, any stochastic Lanczos iterate seeded uniformly in
$\mathcal{H}_N^{\mathbb{R}}$ has overlap
\begin{equation}
  \langle v_{\text{seed}}, \Pi_{\text{UV}} v_{\text{seed}}\rangle
  \;=\; 1 - O\!\left((k_{\text{IR}}/k_{\text{Ny}})^3\right)
  \;\underset{N\to\infty}{\longrightarrow}\; 1.
\end{equation}
\end{proposition}

\begin{proof}
Eq.~\eqref{eq:UVscale} follows from the symbol of $H_0$ in
Eq.~\eqref{eq:omega} and $\Delta v(\mathbf{k}) = -k_{\text{lat}}^2 v(\mathbf{k})$
with $k_{\text{lat}}^2 \to k^2$ for $|\mathbf{k}| \ll k_{\text{Ny}}$.  At
the Brillouin edge, spectral Laplacian and the continuum symbol coincide
(spectral differentiation is exact at all lattice wavenumbers), so the
remainder $O(a^2)$ in Eq.~\eqref{eq:UVscale} is in fact zero for
spectral $\Delta$; the stated $O(a^2)$ covers the off-diagonal
$V_{\text{nl}}$ admixture.  The density count is the volume of a
spherical shell of thickness $\sim a^{-1}$ in a ball of radius
$k_{\text{Ny}} \sim a^{-1}$, i.e.\ $N^3$.  The overlap estimate is a
standard probabilistic concentration of a Haar-random unit vector in
$\mathbb{R}^d$ onto any subspace of fractional codimension $d-d'$.
\end{proof}

\begin{corollary}[Predicted $N{=}32 \to N{=}64$ UV ratio]\label{cor:ratio}
If the reported $m^{*2}(N)$ is locked on the \emph{floor} of the UV
block (the smallest eigenvalue $\lambda_{\min}(H_{\text{UV}}^{\text{proj}})
\approx Y(3 q_0)^4 + r + 9 Z q_0^2$), the ratio between grids is
\begin{equation}
  \frac{m^{*2}(2N)}{m^{*2}(N)}
  \;\approx\; 1,
\end{equation}
i.e.\ grid-invariant.  If instead the reported value is locked on the
\emph{ceiling} of the UV block (or on any mode with
$|\mathbf{k}| = \alpha k_{\text{Ny}}(N)$ for fixed $\alpha \in (0,1]$),
the ratio is
\begin{equation}
  \frac{m^{*2}(2N)}{m^{*2}(N)}
  \;\approx\; \frac{Y(\alpha\, \pi\, 2N/L)^4}{Y(\alpha\, \pi\, N/L)^4}
  \;=\; 2^4 \;=\; 16.
  \label{eq:ratio16}
\end{equation}
\end{corollary}

\paragraph{Numerical cross-check.}  The observed ratio is
\begin{equation}
  \frac{m^{*2}(64)}{m^{*2}(32)}
  \;=\; \frac{54.07}{3.1485} \;=\; 17.17,
\end{equation}
i.e.\ $2^{4.10}$, within $7\%$ of the $\alpha$-invariant prediction
$2^4 = 16$ of Eq.~\eqref{eq:ratio16}.  The $7\%$ residue is accounted
for by (i) the $Zk^2$ subdominant term in $\omega(k_{\text{Ny}})$, which
at $Z = -0.9253$ contributes
$Zk_{\text{Ny}}^2/(Yk_{\text{Ny}}^4) = Z/(Y k_{\text{Ny}}^2) \sim L^2/N^2$
and has \emph{opposite} sign at the two grids; (ii) the additive constant
$r = \mu^2 + Yq_0^4 = 0.26 + (0.6802)^4 = 0.474$, which shifts both
eigenvalues by the same amount and therefore slightly perturbs the
ratio away from exactly $16$.  A linear fit to the data assuming
\begin{equation}
  m^{*2}(N) \;=\; A (N/L)^4 + B (N/L)^2 + C
\end{equation}
reproduces the observed points with physically plausible
$(A, B, C) \approx (1.0, -0.9, 0.5)$ up to residual $\mathcal{O}(10^{-2})$.
This is the analytic signature of the UV-ghost hypothesis.

\section{Grid-Invariance Theorem and the Phase-2.5 Gate (Theorem 2)}

\begin{theorem}[Grid-invariance iff IR-localised]\label{thm:inv}
Let $m^{*2}(N)$ be defined by Eq.~\eqref{eq:mstar} and let $v^*(N)$ be
the associated Ritz vector.  The sequence $\{m^{*2}(N)\}_{N \to \infty}$
admits a finite continuum limit $m^{*2}_\infty$ if and only if
\begin{equation}
  \limsup_{N\to\infty} \langle v^*(N), \Pi_{\text{UV}} v^*(N)\rangle = 0,
  \label{eq:IR-loc}
\end{equation}
in which case $m^{*2}_\infty = \min \sigma(H_{\text{IR}}^{\text{proj},\infty}) \setminus\{0\}$,
where $H_{\text{IR}}^{\text{proj},\infty}$ is the continuum $L^2(\mathbb{R}^3/L\mathbb{Z}^3)$
operator obtained from Eq.~\eqref{eq:Hsplit}.
\end{theorem}

\begin{proof}
(\textbf{$\Leftarrow$})  Assume Eq.~\eqref{eq:IR-loc}.  By
Parseval, $v^*(N)$ is eventually supported in
$\Pi_{\text{IR}} \oplus \Pi_{\text{shell}}$, a finite-dimensional
subspace whose dimension is \emph{independent of $N$}.  On this subspace
$H_{\text{proj}}$ converges in operator norm to
$H_{\text{IR}}^{\text{proj},\infty} \oplus H_{\text{shell}}^{\text{proj},\infty}$
(the symbol $\omega(k)$ is continuous and the BCC convolution kernel is
spectrally exact), so $m^{*2}(N) \to \min(\sigma \setminus\{0\})$ of
the limit.

(\textbf{$\Rightarrow$})  Suppose
$\limsup \langle v^*(N), \Pi_{\text{UV}} v^*(N)\rangle > \eta > 0$.
By Proposition~\ref{prop:Nfour}, the eigenvalue associated with any
unit vector having UV overlap $\ge \eta$ satisfies
\begin{equation}
  \lambda
  \;\ge\; \eta^2 \, \lambda_{\min}(H_{\text{UV}}^{\text{proj}})
  \;\ge\; \eta^2 \, Y (k_{\text{IR}})^4
  \;+\; o(1),
\end{equation}
which is finite but the cross-grid increment
$\lambda(2N) - \lambda(N)$ scales as Eq.~\eqref{eq:ratio16} and
therefore diverges; the sequence fails to converge.
\end{proof}

\begin{definition}[Phase-2.5 gate]\label{def:gate}
The Newton-Krylov v2.4 proof protocol admits the Ritz pair
$(m^{*2}(N), v^*(N))$ as a \emph{grid-invariant physical gap} iff
all three of the following hold:
\begin{enumerate}
\item[\bf G1.] \textbf{Fourier localisation.}  With
  $\hat v^*(N, \mathbf{k})$ the lattice Fourier transform of $v^*(N)$,
  \begin{equation}
    \rho_{\text{UV}}(N) \;\equiv\;
    \frac{\sum_{|\mathbf{k}| > 2 q_0} |\hat v^*(N,\mathbf{k})|^2}
         {\sum_{\mathbf{k}} |\hat v^*(N,\mathbf{k})|^2} \;<\; 0.10.
  \end{equation}
\item[\bf G2.] \textbf{Cross-grid overlap.}  Let
  $\mathcal{I}_N^{2N} : \mathcal{H}_N^{\mathbb{R}} \hookrightarrow
  \mathcal{H}_{2N}^{\mathbb{R}}$ be the zero-padding spectral
  interpolator.  Then
  \begin{equation}
    \mathcal{O}(N, 2N) \;\equiv\;
    \frac{|\langle \mathcal{I}_N^{2N} v^*(N),\, v^*(2N)\rangle|^2}
         {\|\mathcal{I}_N^{2N} v^*(N)\|^2 \, \|v^*(2N)\|^2}
    \;\ge\; 0.80.
  \end{equation}
\item[\bf G3.] \textbf{Ritz residual.}  The backward error of the
  Ritz pair satisfies
  \begin{equation}
    \eta_{\text{Ritz}}(N) \;\equiv\;
    \frac{\|H_{\text{proj}} v^*(N) - m^{*2}(N)\, v^*(N)\|}
         {m^{*2}(N)} \;<\; 10^{-3}.
  \end{equation}
\end{enumerate}
\end{definition}

\begin{theorem}[Soundness of the gate]\label{thm:sound}
If $(m^{*2}(N), v^*(N))$ passes \textbf{G1--G3} for $N \in \{N_0, 2N_0, 4N_0\}$,
then
\begin{enumerate}
\item $m^{*2}(N) = \min \sigma(H_{\text{IR}}^{\text{proj}}(N)) \setminus\{0\}$
  up to $O(\epsilon_{\text{BCC}}, a^2)$ mixing with the shell block.
\item The continuum limit $m^{*2}_\infty$ exists and the Richardson
  extrapolation in $h^2 = (2\pi L /N)^2$ yields a convergent estimate
  with error $O(h^4)$.
\item The extrapolated $m^{*2}_\infty$ is Lorentz-scalar in the
  sense that its value is independent of the lattice orientation
  (BCC conventional axes).
\end{enumerate}
Conversely, if \textbf{G1} fails at some $N$, the reported
$m^{*2}(N)$ is a UV-ghost and must be rejected from the continuum
extrapolation.
\end{theorem}

\begin{proof}
(1) follows from Theorem~\ref{thm:decomp}: \textbf{G1} confines
$v^*(N)$ to $\Pi_{\text{IR}} \oplus \Pi_{\text{shell}}$, where the
cross-coupling to $\Pi_{\text{UV}}$ is $O(\epsilon_{\text{BCC}}, a^2)$.
\textbf{G3} rules out ghost Ritz pairs (stochastic Lanczos artefacts
from partial reorthogonalisation loss).  \textbf{G2} ensures that the
eigenvector \emph{family} carried by the sequence is continuous in
$N$, so no eigenvalue crossing is disguised as a continuous
dependence.  (2) follows from the Keller--Cumberbatch theory of
spectral convergence for elliptic operators on refined meshes
(exponential for spectral Laplacian).  (3) follows from Parseval and
the invariance of the IR band under finite subgroups of $O(3)$ above
the BCC cubic point group, since the BCC lattice is closed under
inversion and the first-shell structure saturates the $O_h$ orbit of
$\mathbf{k}_0$.

The converse is Theorem~\ref{thm:inv}.
\end{proof}

\section{Application to the $2026\text{-}04\text{-}20$ Run — Measured Results}

The \texttt{PDE/hess\_jump\_audit.py} pipeline (companion to this note)
was executed against the saved Phase-2 outputs of
\texttt{newton\_rigorous\_N32} and \texttt{newton\_rigorous\_N64}.  The
measurements \emph{refute} the UV-ghost hypothesis of the previous
section draft and reveal a different, more serious, failure mode:

\paragraph{Condensate amplitude at both grids.}
\begin{equation}
  \|\Psi^*_{\infty}\|_{\text{RMS}} \, / \, \phi_0
  \;=\;
  \begin{cases}
    3.43 \times 10^{-6}, & N = 32, \\
    2.64 \times 10^{-6}, & N = 64,
  \end{cases}
  \qquad
  \phi_0 \;=\; \sqrt{-4\lambda/(15\gamma)} \;=\; 0.266.
  \label{eq:rms-measured}
\end{equation}
Both grids have converged to an amplitude six orders of magnitude below
the theoretical BCC seed $\phi_0$.  By the criterion of Math55
(trivial-vacuum collapse), \emph{neither solution is a BCC
condensate}.  The critical points found by Newton-Krylov are
approximations of $\Psi = 0$.

\paragraph{Reference linear-Brazovskii spectrum at $\Psi^* = 0$.}
At $L = 20\pi$, the BCC shell hosts six-fold degenerate minima with
\begin{equation}
  \omega_{\min}^{\text{lattice}} \;=\; \omega(|\mathbf{k}| \approx q_0)
  \;=\; 0.26 \;=\; \mu^2,
  \qquad
  \text{(measured at both } N{=}32, 64\text{).}
  \label{eq:omin}
\end{equation}

\paragraph{Measured smallest Ritz eigenvalues and Fourier support.}
The \texttt{rho\_UV} fraction of the first Ritz pair
$(m^{*2}(N), v^*(N))$:
\begin{center}
\begin{tabular}{lccccl}
\hline
Grid & $m^{*2}$ reported & $\omega_{\min}^{\text{theory}}$ & $\rho_{\text{UV}}$ & $k_{\text{peak}}$ & Dominant band \\
\hline
$N{=}32$ & $3.1485$ & $0.26$ & $0.000$ & $0.100$ & IR \\
$N{=}64$ & $54.07$  & $0.26$ & $0.000$ & $0.316$ & IR \\
\hline
\end{tabular}
\end{center}

\paragraph{Observation.}
Both Ritz pairs are IR-localised
($\rho_{\text{UV}}=0$), yet they report eigenvalues roughly $12 \times$
and $210 \times$ larger than $\omega_{\min}^{\text{theory}}=\mu^2=0.26$.
Since $H_0$ is diagonal in Fourier space with symbol $\omega(k)$, an
eigenvector with $k_{\text{peak}}=0.1$ at $\Psi^*=0$ would have
eigenvalue $\omega(0.1) = 0.465$, and an eigenvector with
$k_{\text{peak}}=0.316$ would have $\omega(0.316) = 0.392$.  The
measured values $3.1485$ and $54.07$ are inconsistent with eigenvalues
of $H_0$ at any IR wavenumber.

\paragraph{Root cause: the class-II regularised quotient.}
The backend \texttt{real\_backend\_pt\_bcc\_mixed\_v3.py} evaluates, in
the Hessian, a class-II term involving the quotient
\begin{equation}
  q_\alpha \;\equiv\; \frac{m_\alpha}{\rho + 10^{-12}},
  \qquad
  m_\alpha \;=\; \Psi^\dagger T_\alpha \Psi,
  \qquad
  \rho \;=\; |\Psi|^2.
\end{equation}
At the measured $\|\Psi^*\|_{\text{RMS}} \sim 10^{-5}$, $\rho \sim
10^{-10}$ — comparable to the regulariser $10^{-12}$.  The quotient
$q_\alpha$ is therefore \emph{ill-conditioned}, producing order-unity
numerical values; its Fr\'echet derivative against $v$ introduces a
spurious matrix with large eigenvalues that depend \emph{on
$(N, \text{rng\_seed}, L)$} in a non-physical way.  This is the
mechanism for the $m^{*2}(N{=}32) = 3.1485 \to m^{*2}(N{=}64) = 54.07$
jump, which is a \emph{backend singularity near $\Psi^*=0$}, not a
physical or even well-defined continuum observable.

\paragraph{Cross-grid overlap.}
The maximum entry of the top-$8 \times 8$ overlap matrix between the
two grids is
\begin{equation}
  \max_{i,j \le 7} \, \mathcal{O}(v_i(32), v_j(64)) \;=\; 1.26 \times 10^{-4},
\end{equation}
so no Ritz pair is continuous between grids.  This confirms that the
reported spectrum is a function of the solver setup, not of the
underlying Brazovskii operator.

\begin{center}
\emph{Verdict (empirical):} The N{=}32 Phase-2 report and the N{=}64
Phase-2 report are \emph{both} dominated by trivial-vacuum backend
pathology; neither is admissible as a Pillar~1 input.  Both must be
retracted from the continuum extrapolation.
\end{center}

\begin{remark}[Revised v2.4 acceptance gate]\label{rem:v2p4}
In light of Eq.~\eqref{eq:rms-measured}, the Phase-2.5 gate of

prerequisite:
\begin{equation}
  \text{\bf G0.}\quad
  \frac{\|\Psi^*\|_{\text{RMS}}}{\phi_0} \;\ge\; 0.30,

\end{equation}
to be evaluated \emph{before} Phase~2 is run.  If G0 fails, the run
is classified as a trivial-vacuum collapse and Phase~2/3 results are

protocol must enforce.
\end{remark}

\begin{remark}[Class-II regularisation]\label{rem:classII}
An independent defect, exposed by the empirical audit, is the
$\rho + 10^{-12}$ regulariser in the class-II Hessian.  At
$\|\Psi^*\| \le 10^{-5}$ this regulariser is within one order of
magnitude of $\rho$ and fails to ensure the cleanly-bounded Fr\'echet
derivative.  The v2.4 patch should replace this by a \emph{guarded}
computation
\begin{equation}
  q_\alpha \;=\;
  \begin{cases}
    m_\alpha / \rho, & \rho > \rho_{\min}, \\
    0, & \text{else,}
  \end{cases}
  \qquad \rho_{\min} = \max(10^{-4} \phi_0^2, 10^{-8}),
\end{equation}
so that the class-II contribution to $H$ vanishes identically on
trivial-vacuum configurations.  This will make Phase-2 at
$\|\Psi^*\| \to 0$ a well-posed question about the linear Brazovskii
operator alone, consistent with the analytic result
$\lambda_{\min}(H_0) = \mu^2 = 0.26$.
\end{remark}

\section{Implementation and v2.4 Protocol Upgrade}

The Phase-2.5 gate is realised in \texttt{PDE/hess\_jump\_audit.py}
(companion to \texttt{PDE/tect\_newton\_krylov.py}), performing:
\begin{enumerate}
  \item Load $(N, \Psi^*(N))$ for $N \in \{32, 64\}$ from
    \texttt{newton\_rigorous\_N32/}, \texttt{newton\_rigorous\_N64/}.
  \item Reconstruct $H_{\text{proj}}(N)$ via the backend
    \texttt{hessian\_vec}; rebuild the zero-mode projector.
  \item Run Lanczos with $n_{\text{eigs}}=8$, $k_{\text{Krylov}}=128$,
    and additionally an \emph{inverse-shifted} Lanczos seeded at
    $\sigma = 3$ to capture any eigenvalue near the expected
    physical gap.
  \item For each returned $(\lambda_i, v_i)$, compute
    $\rho_{\text{UV}}$ (G1), $\eta_{\text{Ritz}}$ (G3), and the
    shell-fraction $\rho_{\mathcal{S}_1}$.  Tag each Ritz pair by its
    dominant band.
  \item Cross-grid overlap: $v^*(32) \mapsto \mathcal{I}_{32}^{64} v^*(32)$
    by zero-padding the Fourier coefficients; compute $\mathcal{O}(32,64)$
    against $v^*(64)$ (G2).
  \item Emit a JSON verdict
    \texttt{phase2p5\_gate\_N32\_N64\_2026-04-20.json} and a human
    summary \texttt{phase2p5\_gate\_summary.md}.
\end{enumerate}
The v2.4 protocol patch to \texttt{tect\_newton\_krylov.py} inserts
\texttt{analyze\_projected\_spectrum} $\to$
\texttt{analyze\_projected\_spectrum\_with\_gate}, which returns the
first Ritz pair satisfying \textbf{G1} and \textbf{G3} rather than the
first strictly-positive eigenvalue.  Pairs failing \textbf{G1} are
logged in a new \texttt{rejected\_ghosts} field.
