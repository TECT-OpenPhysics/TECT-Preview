\section*{3. The analytic removal pathway (new)}

All symbols as in Math437 v1.2: $r = \mu^2 = 0.005$, $u = -0.86$,
$v = 3.24$, $M(m) = \int \frac{d^3k}{(2\pi)^3}\,
\bigl[m + C(k^2 - q_0^2)^2\bigr]^{-1}$ (cell-averaged variance of the
trial dressing), $M_c = -u/(10v) = 43/1620 = 0.0265432$,
$u_{\rm eff}(M) = u + 10vM$, and the Math437 stationarity identity
\[
\frac{dF_0}{dm} \;=\; \tfrac12\,M'(m)\,g(m),
\qquad
g(m) \;:=\; r - m + 3u\,M(m) + 15v\,M(m)^2 ,
\]
with $g(m^*) = 0$ being EXACTLY the gap equation.

\begin{lemma}[L1: strict monotonicity of $M$]
$M'(m) = -\int \frac{d^3k}{(2\pi)^3}\,[m + C(k^2-q_0^2)^2]^{-2} < 0$
for every $m > 0$. (Differentiation under the integral: for $m \ge m_0
> 0$ the integrand and its $m$-derivative are dominated by the
integrable $m_0$-versions --- large-$k$ decay $k^{-2}$ and $k^{-6}$
respectively in the radial measure; this textbook regularity step is
the named gap G-A0-DUI below, expected routine.)
\end{lemma}

\begin{lemma}[L2: $g$ is strictly decreasing wherever $u_{\rm eff}\ge0$]
$g'(m) = -1 + 3\,u_{\rm eff}(M(m))\,M'(m)$. On the region $M(m) \ge
M_c$ (equivalently $m \le m_c$ where $M(m_c) = M_c$): $u_{\rm eff} \ge
0$ and $M' < 0$ give $g'(m) \le -1 < 0$.
\end{lemma}

\begin{lemma}[L3: $g < 0$ beyond the window, closed-form]
On $M(m) < M_c$ (i.e.\ $m > m_c$): using $u < 0$ (so $3uM < 0$) and
$0 \le M < M_c$,
\[
g(m) \;<\; r + 15v\,M_c^2 - m \;=\; 0.005 + 48.6\times(43/1620)^2 - m
\;=\; 0.039241 - m .
\]
Hence $g(m) < 0$ for every $m > m_w := 0.039241$.
\end{lemma}

\begin{theorem}[A-zero uniqueness and zero-at-gap, conditional skeleton]
Assume (i) L1 (modulo G-A0-DUI), and (ii) the ANCHOR WINDOW INEQUALITY
$m^* > m_w$. Then $F_0$ is strictly decreasing on $(0, m^*)$ and
strictly increasing on $(m^*, \infty)$; consequently $\Delta F_0(m) >
0$ for all $m \ne m^*$ and $\Delta F_0(m^*) = 0$ --- i.e.\ H-A0's (U)
and (Z) hold as a THEOREM, with no curve-shape numerics.
\end{theorem}

\begin{proof}
Since $M(m^*) = M_R > M_c$ (the recorded anchor inequality, $\times4.1$
margin), $m^* < m_c$. On $(0, m_c]$: L2 gives $g$ strictly decreasing;
$g(m^*) = 0$ forces $g > 0$ on $(0, m^*)$ and $g < 0$ on $(m^*, m_c]$,
and in particular $m^*$ is the UNIQUE zero of $g$ on $(0, m_c]$. On
$(m_c, \infty)$: $m > m_c \ge m^* > m_w$ --- wait, the chain needed is
$m > m_c$ and $m_c > m_w$; the latter follows from $m_c > m^* > m_w$
(window inequality) --- so L3 gives $g < 0$ there; no further zero.
By L1, $\mathrm{sign}(dF_0/dm) = -\mathrm{sign}(g)$: $F_0' < 0$ on
$(0, m^*)$ and $F_0' > 0$ on $(m^*, \infty)$. Strict global minimum at
$m^*$, value $0$ after subtraction, monotone flanks. $\square$
\end{proof}

\noindent\textbf{The anchor window inequality (closed-form arithmetic,
$\times7.8$).} The gap equation at the production anchors gives
\[
m^* \;=\; r + 3u M_R + 15v M_R^2
\;=\; 0.005 - 2.58\times0.109414 + 48.6\times0.109414^2
\;=\; 0.304523 ,
\]
which reproduces the certified artefact anchor $r_R = 0.3045257$ to
$5\times10^{-6}$ (the residual is $M_R$ display rounding) --- so $m^* =
r_R$ identically by the gap equation, and
\[
\frac{m^*}{m_w} \;=\; \frac{0.30453}{0.039241} \;=\; 7.76 .
\]
The window inequality holds with $\times7.8$ closed-form margin at the
production anchors.

VALID-with-mitigation: the identity is the variational chain rule of

recorded in the certified Math437 v1.2 with the gap-equation zero

alongside the arithmetic asserts; until then the pathway is honestly T3.
\item[($\beta$)] \emph{``L3's window uses $M < M_c$ but bounds $15vM^2$
by $15vM_c^2$ --- if $M$ could exceed $M_c$ on $m > m_c$ the bound
fails.''} DISMISSED: $m > m_c \Rightarrow M(m) < M(m_c) = M_c$ is
exactly L1's strict monotonicity (the region is DEFINED by the
inequality); and $M \ge 0$ holds because $M$ is an integral of a
positive integrand. Both directions of the squeeze are structural.
\item[($\gamma$)] \emph{``The proof needs $m^* < m_c$ AND $m_c > m_w$;

m_c$ itself certified?''} DISMISSED with the recorded anchor
inequality: $M(m^*) = M_R = 0.109414 > M_c = 0.0265432$ ($\times4.1$,
quadrature-free at the anchors per the H-A0 row itself), and L1's
monotonicity converts it to $m^* < m_c$. The chain $m_c > m^* > m_w$
then closes with the $\times7.8$ window margin.
\item[($\delta$)] \emph{``This note could be read as discharging H-A0
