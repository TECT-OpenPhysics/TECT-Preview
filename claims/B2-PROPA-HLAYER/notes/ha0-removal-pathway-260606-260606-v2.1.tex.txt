\section*{2. What H-A0 asserted, and the reduction}

H-A0 asserted (U) the $A=0$ curve $F_0(m)$ has a unique minimum at the
gap point $m^*$, (Z) of value $0$ after subtraction, certified
numerically on a consistent quadrature scheme (with a recorded
$5.5\times10^{-3}$ scheme-gap systematic). The sign decomposition below
DERIVES (U) and (Z) from structure, so only the anchor inequality
remains as an assumption.

With $r=\mu^2=0.005$, $u=-0.86$, $v=3.24$,
$M(m)=\int\frac{d^3k}{(2\pi)^3}[m+C(k^2-q_0^2)^2]^{-1}$,
$M_c=-u/(10v)=43/1620$, $u_{\rm eff}(M)=u+10vM$, and the Math437
stationarity identity $dF_0/dm=\tfrac12 M'(m)\,g(m)$,
$g(m)=r-m+3uM+15vM^2$ (so $g(m^*)=0$ is the gap equation):

\begin{lemma}[L1]
$M'(m)=-\int\frac{d^3k}{(2\pi)^3}[m+C(k^2-q_0^2)^2]^{-2}<0$ for $m>0$
(differentiation under the integral; integrand and $m$-derivative
dominated by integrable $k^{-4},k^{-6}$ tails for $m\ge m_0>0$ --- the
textbook residual G-A0-DUI).
\end{lemma}
\begin{lemma}[L2]
$g'(m)=-1+3u_{\rm eff}(M(m))M'(m)\le -1<0$ wherever $M(m)\ge M_c$
(i.e.\ $m\le m_c$), since $u_{\rm eff}\ge0$ and $M'<0$.
\end{lemma}
\begin{lemma}[L3]
On $M(m)<M_c$ (i.e.\ $m>m_c$), $u<0$ and $0\le M<M_c$ give
$g(m)<r+15vM_c^2-m = m_w - m$, $m_w:=0.0392407$; hence $g<0$ for
$m>m_w$.
\end{lemma}
\begin{theorem}[A-zero uniqueness, conditional on H-ANCHOR + G-A0-DUI]
Assume L1 (modulo G-A0-DUI) and the anchor inequality $m^*>m_w$
(H-ANCHOR; with $M_R>M_c$ giving $m^*<m_c$). Then $g>0$ on $(0,m^*)$ and
$g<0$ on $(m^*,\infty)$ (L2 on $(0,m_c]$, L3 on $(m_c,\infty)$, glued at
$m_w<m^*<m_c$), so by L1 $F_0'<0$ on $(0,m^*)$ and $F_0'>0$ on
$(m^*,\infty)$: a strict global minimum at $m^*$, value $0$ after
subtraction. H-A0's (U) and (Z) hold as a theorem, with no curve-shape
numerics and no quadrature-scheme dependence.
\end{theorem}
