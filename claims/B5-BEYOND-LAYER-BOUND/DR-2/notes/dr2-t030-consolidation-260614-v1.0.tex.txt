\section*{2. The three precise versions of T-030}

$N=|Q|$, $E_+(Q)=\#\{(a,b,c,d)\in Q^4:a+b=c+d\}$.

\paragraph{T-030a (lattice, $R^\varepsilon$-loss) -- CLOSED conditionally on R-026.}
\[
Q\subset\mathbb Z^3\cap\{|x|^2=R\}\ \Longrightarrow\ E_+(Q)\ \ll_\varepsilon\ R^\varepsilon N^2.
\]
Source: R-026, assembled as the self-contained Theorem in Sec.~A.4 (T7-NTstandard --
a theorem modulo the textbook Dirichlet
class-number formula and the divisor bound only. \emph{External positioning:}
A.\ Mudgal, \emph{Additive energies on spheres}, JLMS 106(4):2927--2958 (2022),
arXiv:2105.06925, studies exactly $A\subset\mathbb Z^d\cap\{|x|^2=m\}$ additive
energy ($d=3,4$); $d=4$ gives $\ll_\varepsilon m^\varepsilon|A|^{2+1/3-1/1392}$,
$d=3$ improves Benatar--Maffucci. Mudgal confirms the object is studied and
controlled, but is NOT the source of the clean $R^\varepsilon N^2$ here -- that is
R-026 (Sec.~A.3); no specific Mudgal $d=3$ theorem number is attributed.

\paragraph{T-030b (lattice, $N^2$ polylog) -- OPEN.}
$Q\subset\mathbb Z^3\cap\{|x|^2=R\}\ \overset{?}{\Longrightarrow}\ E_+(Q)\lesssim N^2\log^{O(1)}N$:
the $\varepsilon$-removal $R^\varepsilon\to\log^{O(1)}|Q|$, strictly stronger than
T-030a, not found as a known theorem.

\paragraph{T-030-real (arbitrary real $Q\subset S^2$) -- OPEN (Bourgain--Demeter).}
$Q\subset S^2\ \overset{?}{\Longrightarrow}\ E_+(Q)=E_2(Q)\lesssim_\varepsilon(\#Q)^{2+\varepsilon}$:
a Bourgain--Demeter conjecture (exposited by Sheffer). As of Sheffer's 2016
exposition the best reported bound for $E_2$ on $S^2$ was $O(N^{7/3})$ -- cited as
the 2016 state, NOT a current-best claim (the literature may have advanced). A
DISTINCT object, the third-order energy $E_3$ on $S^1$, has best-known

\section*{Appendix A. Self-contained proof of T-030a (with the B5-closure lemmas)}

\textbf{A.1 Lemma A (R-025; T7; full proof: dr2-circle-richness-reduction v1.0).}
For finite $Q\subset S^2$, $E_+(Q)\le(1+T'(Q))N^2$ where
$T'(Q)=\max_{m\neq0,\,r(m)>0}\#(Q\cap C_m)$, $C_m=S^2\cap\{x\cdot m=|m|^2/2\}$.
\emph{Proof.} For $m\neq0$, $a+b=m$ forces $a,b\in C_m$, so $r(m)\le\#(Q\cap C_m)\le
T'$; hence $\sum_{m\neq0}r(m)^2\le T'\sum_{m\neq0}r(m)\le T'N^2$, and $r(0)^2\le N^2$.
Elementary, decoupling-free. $\square$

\textbf{A.2 Lemma 2 (R-032; T7; full proof: coherence-offdiag-comfortable-bound v1.0).}
For a $\theta_{\min}$-separated $Q$ on a sphere of radius $Q_0$, every proper
sum-circle $C_t$ (radius $\rho=\sqrt{Q_0^2-|t|^2/4}\le Q_0$) has
$\#(Q\cap C_t)\le\lfloor2\pi/\theta_{\min}\rfloor$.
\emph{Proof.} Points on $C_t$ are pairwise Euclidean $\ge2Q_0\sin(\theta_{\min}/2)$;
$k$ points pairwise $\ge d$ on a radius-$\rho$ circle satisfy
$k\le\pi/\arcsin(d/2\rho)$, and $\rho\le Q_0$ gives $d/2\rho\ge\sin(\theta_{\min}/2)$,
so $k\le2\pi/\theta_{\min}$. For the TECT admissible window this is $10$. $\square$

\textbf{A.3 Lattice sum-circle occupancy [DIV-CIRC] (R-026; T7-NTstandard; full
proof: dr2-lattice-divisor-closure v1.2).} For $m\in\mathbb Z^3$ with
$0<|m|^2<4R$ (so $R':=4R-|m|^2>0$),
\[
\#(\mathbb Z^3\cap C_m)\ \le\ 6\,d(R')\ \ll_\varepsilon R^\varepsilon,
\]
uniformly in $m$ ($d$ the divisor function); the degenerate boundary
$|m|^2=4R$ ($R'=0$) gives $\#(\mathbb Z^3\cap C_m)\le1$ (case (i) below).
\emph{Proof.} \emph{(i) Shift removal.} $C_m$ has centre $m/2$. For
$x\in\mathbb Z^3\cap C_m$ set $y:=2x-m\in\mathbb Z^3$; then
$y\cdot m=2x\cdot m-|m|^2=0$, so $y$ lies in the rank-$2$ sublattice
$\Lambda_m:=\mathbb Z^3\cap m^{\perp}$, and
$|y|^2=4|x|^2-4x\cdot m+|m|^2=4R-|m|^2=:R'$. The map $x\mapsto y$ is injective
($x=(y+m)/2$), so $\#(\mathbb Z^3\cap C_m)\le r_Q(R'):=\#\{y\in\Lambda_m:|y|^2=R'\}$,
a HOMOGENEOUS representation count by the positive-definite integral binary
quadratic form $Q(y)=|y|^2$ on $\Lambda_m$ (the centre $m/2$ is removed exactly).
If $R'=0$ (i.e.\ $|m|^2=4R$, the tangent-plane degeneracy) then $y=0$ is the only
solution and, by injectivity, $\#(\mathbb Z^3\cap C_m)\le1$; so assume henceforth
$0<|m|^2<4R$, whence $R'>0$ and $d(R')$ is defined.
\emph{(ii) Class-number bound.} For a primitive positive-definite integral binary
form of discriminant $\Delta$, Dirichlet's class-number formula gives
$\sum_{[f]\,\mathrm{disc}\,\Delta}r_f(n)=w(\Delta)\sum_{d\mid n}\chi_\Delta(d)\le6\,d(n)$
($w(\Delta)\le6$, $\chi_\Delta\le1$); a single class is bounded by the sum over all
classes, so $r_Q(R')\le6\,d(R')$ (an imprimitive content $g$ reduces to
$r_{Q'}(R'/g)\le6\,d(R')$). This is UNIFORM in $m$ (independent of $\Delta$, hence
of the plane). \emph{(iii) Divisor bound.}
$d(R')\ll_\varepsilon R'^{\varepsilon}\le(4R)^{\varepsilon}\ll_\varepsilon R^\varepsilon$.
$\square$ Both inputs -- the class-number formula and
$d(n)\ll_\varepsilon n^\varepsilon$ -- are textbook, so [DIV-CIRC] is a theorem,
not a hypothesis.

\textbf{A.4 Theorem (T-030a), self-contained modulo textbook number theory.} For
$Q\subset\mathbb Z^3\cap\{|x|^2=R\}$,
\[
E_+(Q)\ \le\ (1+C_\varepsilon R^\varepsilon)N^2\ \ll_\varepsilon\ R^\varepsilon N^2.
\]
\emph{Proof.} By A.1 (Lemma A), $E_+(Q)\le(1+T'(Q))N^2$ with
$T'(Q)=\max_{m\neq0,\,r(m)>0}\#(Q\cap C_m)$. Each contributing $m=a+b$
($a,b\in Q$) has $0<|m|^2\le4R$; by A.3 its occupancy is $\le1$ when $|m|^2=4R$
and $\le6\,d(4R-|m|^2)$ when $0<|m|^2<4R$, so
\[
T'(Q)\le\max\Big\{1,\ 6\max_{0<|m|^2<4R}d(4R-|m|^2)\Big\}\ll_\varepsilon R^\varepsilon.
\]
Substituting gives the claim. $\square$ This is the frozen in-repo certificate for T-030a: it uses
Lemma A (A.1) and the two textbook number-theoretic inputs of A.3 -- no decoupling,
no incidence conjecture, no external theorem. (Mudgal 2022 corroborates the object
but is not used in this proof.)
