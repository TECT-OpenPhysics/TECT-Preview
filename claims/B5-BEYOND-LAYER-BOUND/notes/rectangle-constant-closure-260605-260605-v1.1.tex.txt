\section*{2. Theorem-grade rectangle constant (operator derivation, verdict \#9)}

\begin{proposition}[Rigorous $\sqrt n$ energy bound; operator-supplied]
Let $N = 2n = |\hat Q|$ and let $p_C$ be the ordered antipodal pair count
of carrier $C$. Three distinct points lie on at most one circle, so
\[
\sum_{C:\,k_C \ge 3} p_C^3 \le \sum_{C:\,k_C\ge3} k_C^3
\le 27\binom{N}{3} < \tfrac{9}{2}N^3,
\qquad
\sum_{C:\,p_C \le 2} p_C^3 \le 4\sum_C p_C \le 4N^2,
\]
hence $\sum_C p_C^3 \le 7N^3$ conservatively. With $\sum_C p_C \le N^2$
and Cauchy--Schwarz interpolation,
\[
\sum_C p_C^2
\le \Bigl(\sum_C p_C\Bigr)^{1/2}\Bigl(\sum_C p_C^3\Bigr)^{1/2}
\le \sqrt7\, N^{5/2}.
\]
For $\kappa$-balanced patterns ($A_{\max}^2 \le \kappa^2 I/n = 2\kappa^2
I/N$), $\Psi_C \le p_C A_{\max}^2$ gives
\[
\sum_{t\ne0} w_t^2 \;\le\; \lambda'^2 I^2
\Bigl(8 + 4\sqrt{14}\,\kappa^4 \sqrt n\Bigr),
\qquad
c_R^{\rm rigorous} = 4\sqrt{14} \simeq 14.97 .
\]
The v2.0 calibrated corollary is hereby THEOREM-grade; the certified
region at the anchor budget ($K_{\rm budget} = 5972$ at
$I = 4\times10^{-4}$) is $n \lesssim 1.59\times10^{5}$
($3.77\times10^3$ at $10^{-3}$; $2.02\times10^2$ at $2\times10^{-3}$).
Machine checks: $\sum p^3 \le 7N^3$ and the interpolation hold on all test
configurations.
\end{proposition}

\section*{3. New incidence route: exponent $28/13$}

\begin{theorem}[Stereographic transfer]
A stereographic projection from a pole not on any carrier maps the sphere
circles bijectively to circles (or lines) in the plane, preserving all
point--circle incidences. (Machine check: projected ring circles are exact
plane circles, residuals $\sim 10^{-30}$.)
\end{theorem}

\begin{theorem}[Incidence upgrade --- repaired exponent $20/9$]
Let $m_r$ be the number of carriers with $p_C \ge r$. The pair partition
gives the cap $m_r \le 2N^2/r$; the planar Aronov--Sharir-type incidence
bound gives $m_r \le C\bigl(N^2/r^3 + N^3 L^{11/2}/r^{11/2} + N/r\bigr)$
($L$ polylogarithmic). The pair cap and the $r^{-11/2}$ term cross at
\[
2N^2/r_1 = N^3 L^{11/2}/r_1^{11/2}
\;\Longleftrightarrow\;
r_1 = (N L^{11/2}/2)^{2/9} ,
\]
so the dyadic sum is dominated by its value at $r_1$:
\[
\sum_C p_C^2 \;=\; \sum_r m_r\, r^2
\;\le\; 2N^2 r_1 + O\!\Bigl(\frac{N^3 L^{11/2}}{r_1^{7/2}}\Bigr)
\;=\; O\!\bigl(N^{20/9}\,\mathrm{polylog}\,N\bigr),
\qquad
\tfrac{20}{9} \simeq 2.222 < \tfrac52 .
\]
(The $28/13$ exponent of v1.0 was an arithmetic slip --- caught by the
operator, verify-loop catch \#7; a dyadic self-check at $N = 4096$
confirms the $20/9$ form with $O(1)$ ratio.) Consequently, for
$\kappa$-balanced patterns, $K(n) \le 8 + c\,\kappa^4 n^{2/9}\,
\mathrm{polylog}\,n$; with the conservative $c\,\mathrm{polylog} = 30$
the anchor budget certifies $n \lesssim 2.2\times10^{10}$. This route is
PROVISIONAL (supporting-candidate status) until the Aronov--Sharir
constant is pinned; the OFFICIAL tier basis is the $\sqrt n$ route of
the proposition above.
\end{theorem}

\medskip
\noindent\textbf{3b. Designated route to the sharp bound:
structure vs randomness.} The measured exponents ($\sim n^{2.05}$) and
the unit-circle-strength incidence conjecture both point at
$\sum_C p_C^2 = O(N^2\,\mathrm{polylog})$. A genuinely new route, not
passing through open incidence conjectures, is registered as the
designated multi-turn attack:
\begin{itemize}
\item \emph{Lemma DR-1 (circle fiber rigidity = no small doubling on
circles).} For $A$ contained in a circle, distinct unordered pairs have
distinct sums outside the axial direction (fiber $\le 2$ plus the axial
exception), so $|A + A| \ge c\,|A|^2$ --- subsets of circles can NEVER
have small additive doubling. PROVED in substance by the fiber analysis
of the universal single-circle theorem (main chain v1.8).
\item \emph{Lemma DR-2 (sphere structure dichotomy --- OPEN, multi-turn).}
If $E_2(\hat Q) \ge K N^2$ with $K$ large, Balog--Szemer\'edi--Gowers
yields $A' \subset \hat Q$, $|A'| \gtrsim N/\mathrm{poly}(K)$, with
doubling $|A' + A'| \le \mathrm{poly}(K)\,|A'|$. The dichotomy to prove:
every sphere subset either has $|A+A| \ge |A|^{1+\delta}$ or concentrates
its mass on $O(1)$ circles. Combined with DR-1 (circles forbid small
doubling) this is self-improving and would force
$E_2 = O(N^2\,\mathrm{polylog})$ unconditionally.
\item \emph{Status.} DR-2 is the single open ingredient; it is a
Freiman-type structure statement for positively curved surfaces, related
to (but not identical with) sum-set growth on strictly convex curves
(Elekes--Nathanson--Ruzsa). Registered as OPEN with this note; no closure
quote depends on it.
\end{itemize}

\begin{itemize}

circles; carriers live on the sphere.''} DISMISSED: stereographic
projection from a generic pole is a bijection sphere-circles $\to$
plane-circles/lines preserving incidences (machine-verified exact); the
planar theorem applies verbatim to the projected family.
\item[($\beta$)] \emph{``The incidence exponent itself was wrong in
v1.0.''} UPHELD and repaired: the operator caught the $28/13$ arithmetic
slip (verify-loop catch \#7) --- the correct crossover gives $20/9$. The
lesson is registered: exponent optimizations must carry a numerical
dyadic self-check (now in the script, ratio $1.1$ at $N = 4096$). The
route is demoted to PROVISIONAL supporting-candidate; no ledger figure
uses it.
\item[($\gamma$)] \emph{``H-KBAL ($\kappa$-balance) may exclude physical
competitors.''} VALID-with-mitigation: heavily unbalanced patterns are
dominated by their heavy modes and fall into the small-$n_{\rm eff}$
certified region (the heavy sub-pattern carries the energy); the rigorous
amplitude-dyadic lift is registered. No admissible competitor class is
known to violate H-KBAL at fixed $I$.
\item[($\delta$)] \emph{``The sharp $O(n^2)$ claim rests on three
families.''} DISMISSED as stated: the $O(n^2)$ statement is registered as
a CONJECTURE with a pre-registered falsification gate (exponent $\ge 2.3$
on any family), not as a theorem; no closure quote depends on it.
\end{itemize}
