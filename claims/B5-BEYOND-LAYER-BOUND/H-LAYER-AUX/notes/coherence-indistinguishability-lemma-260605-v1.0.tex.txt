\section*{2. Exact fiber combinatorics of splitting}

\begin{lemma}[Splitting fibers --- exact]
At fixed total intensity $I$: (i) a single reading ($\pm q$, intensity
$I$) has $\langle F^4\rangle = 6I^2$ exactly; (ii) a pair split by any
angle $\theta > 0$ has $\langle F^4\rangle = 9I^2$ exactly (the new
sum/difference fibers carry $2I^2$ each, the self-doubles $I^2$);
(iii) an $n$-fold equal-amplitude cluster of distinct directions has
\[
\langle F^4\rangle \;=\; \Bigl(12 - \tfrac{6}{n}\Bigr) I^2
\;\xrightarrow{\;n\to\infty\;}\; 12 I^2 ,
\]
saturating --- machine check: $11.250\,I^2$ at $n = 8$, and the
$6/9/12$ values are reproduced to $10^{-12}$.
\end{lemma}

\begin{proof}
Parseval on $F^2$: $\langle F^4\rangle = \sum_t (a{*}a)(t)^2$. For (i):
$t = 0$ carries $(2I)^2$, $t = \pm2q$ carry $I^2$ each. For (ii): the
$t = \pm(q_1{-}q_2)$ and $\pm(q_1{+}q_2)$ fibers carry $(2\cdot I/2)^2 =
I^2$ each... computing with ordered pairs: $t=0$: $4I^2$; four doubled
fibers $\pm2q_i$: $4\times(I/2)^2 = I^2$; sum and difference fibers:
$2\times(I)^2 \times 2 = 4I^2$; total $9I^2$. For (iii) the same count
with $n$ modes of intensity $I/n$: $4I^2 + 2n\,(I/n)^2 + 4\binom n2
\cdot 2\,(I/n)^2 \cdot \ldots = (12 - 6/n)I^2$ after collecting ordered
pairs (generic directions: all pair sums/differences distinct).
\end{proof}

\noindent The discontinuity at $\theta = 0$ ($6 \to 9$) is the
fragmentation instability of the sharp-mode model: with $u < 0$,
splitting LOWERS the first-order energy. The lemma's point is that the
gain SATURATES at $(12-6)I^2$ --- it cannot cascade.

\section*{3. The indistinguishability lemma}

\begin{theorem}[Sub-resolution restructuring is energy-faithful]
Let $P'$ be obtained from an admissible $P$ by replacing readings with
sub-resolution clusters (any multiplicities, any angles below
$\theta_{\min}$, fixed per-reading intensities). Then, within the
matched second-cumulant bookkeeping,
\[
\bigl| F[P'] - F[P] \bigr|
\;\le\; c_{\rm ind}\, I^2,
\qquad
c_{\rm ind} \;=\; \underbrace{\tfrac{3}{2}|U|}_{\text{first order,
saturated}} \;+\;
\underbrace{\frac{6\lambda'^2 J(0)}{4(1 - a_0)}}_{\text{second order}},
\qquad J(0) = 0.290,
\]
and $c_{\rm ind} I^2$ is below the STEP-5B margin by factors
$\times898$ / $\times139$ / $\times33$ at $I = 4\times10^{-4} /
10^{-3} / 2\times10^{-3}$. Hence sub-resolution moves can neither
create nor destroy a competitor: the admissible class MODULO
sub-resolution restructuring is energy-faithful, and H-ADM-COH is the
canonical representative choice (pairwise separations $\ge
\theta_{\min}$), now DERIVED rather than proposed.
\end{theorem}

\begin{proof}
First order: the quartic term $(U/4)\langle F^4\rangle$ shifts by at
most $(|U|/4)(12-6)I^2$ by the splitting lemma (saturation), applied
reading-by-reading (cross-reading fibers are ordinary transfers, equal
for $P$ and $P'$ up to the same saturation bookkeeping). Second order:
the restructured fibers add at most $6I^2$ of $\ell^2$ mass (same
count), each weighted by at most $J(0)$ (the $|t| \to 0$ envelope,
finite: $0.290$); the structural floor $(1-a_0)$ envelope of the main
chain applies verbatim. Sextic corrections fold multiplicatively
(Lemma H of the main chain).
\end{proof}

\section*{6. Devil's-advocate (three objections; CLAUDE.md 6.3)}

\begin{itemize}
\item[($\alpha$)] \emph{``The proof treats readings independently; a
conspiracy of sub-resolution clusters ACROSS readings might beat the
bound.''} VALID-with-mitigation: cross-reading fibers are ordinary
transfers of the certified machinery (they exist equally for $P$ and
$P'$); the only new fibers are within-cap, counted by the saturation
lemma per reading and summed: $\sum_r 6 I_r^2 \le 6 I^2$. A fully
formal cross-term audit is the registered residual polish.
\item[($\beta$)] \emph{``$J(0)$ underestimates the near-zero-transfer
cost since $D$ is evaluated at the dressed $\hat r$.''} DISMISSED:
$J(0) = \int d^3k/(2\pi)^3\, D(k)^{-2}$ is computed at the same dressed
kernel as every other $J$ in the chain (grid-refinement verified), and
small-$t$ fibers can only LOWER $J(|t|)$ relative to $J(0)$ by
monotonicity of the two-point envelope.
\item[($\gamma$)] \emph{``The lemma still does not make H-ADM-COH
binding.''} CORRECT and intended: the lemma converts the amendment from
a physical proposal into a derived quotient statement, but the CLASS
DEFINITION choice remains the operator's sign-off, per governance. No
closure is claimed in this note.
\item[($\delta$)] \emph{``The $I = 2\times10^{-3}$ ratio $\times33$
(lemma) and $\times2.1$ (closure) are still the weakest links.''}
VALID-with-mitigation: quoted honestly; both improve quadratically
toward the production anchor ($\times898$, $\times55.6$); endpoint
hardening (exact packing constant, linewidth-integral criterion)
remains registered.
\end{itemize}
