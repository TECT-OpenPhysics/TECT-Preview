\section*{1. Purpose (roadmap Step 1, CLOSED)}

Step 1 certifies the natural boundaries of the Reading-H operating window, replacing the
single operating point $(I,\mu^2)=(2\times10^{-3},0.005)$ by an explicit certified region.
This widens the \emph{parameter} scope; it does NOT widen the competitor class (Step 2, EXT).
No tier change. Operator verdict 2026-06-10: ACCEPTED AS STEP 1 CLOSURE.

\section*{2. Two boundaries; the selection boundary binds}

\[
\textbf{(A) off-diagonal:}\quad R_{\rm lead}=23.2\,(1+K_{\rm floor})\,I<1
\ \Rightarrow\ I<I_c^{\rm off}=\tfrac{1}{23.2\cdot14}=3.08\times10^{-3}\ (K_{\rm floor}\le13),
\]
\[
\textbf{(S) selection:}\quad \mathrm{joint}(\mu^2,I)>1,\qquad
\mathrm{joint}=\frac{\mathrm{MARGIN}}{D},\quad
D=(1+R_{\max})C_2+\frac{\mathrm{MARGIN}-C_2}{S_{\rm un}} .
\]
Numerically $I_c^{\rm sel}=2.41\times10^{-3}<I_c^{\rm off}=3.08\times10^{-3}$ (both $\times10^{-3}$),
so the \textbf{selection boundary binds}, and
\[
\boxed{\ I_c=\min_{\mu^2\in[\times0.5,\times2]}I_c^{\rm sel}(\mu^2)=2.41\times10^{-3}\ }
\]
(at the low-$\mu^2$ edge, $I_c^{\rm sel}$ being increasing in $\mu^2$). The operating endpoint
$I=2\times10^{-3}$ is interior with headroom $(I_c-I)/I=20\%$. $I_c$ is a PHYSICAL selection
phase boundary (joint$\to1$ = third-cumulant degeneracy), not a proof gap.

\section*{3. The widened mu^2 window: to the physical branch boundary}

$\mathrm{MARGIN}(\mu^2)=\mathrm{PB}(M_+)-\mathrm{DIP\_BAND}$ requires the upper PD branch
$M_+=(-3U+\sqrt{9U^2-60V\mu^2})/(30V)$, i.e.\ $\mathrm{disc}=9U^2-60V\mu^2>0$:
\[
\mu^2<\mu^2_{\max}=\frac{3U^2}{20V}=0.0342 .
\]
Across $(0,\mu^2_{\max})$: $\mathrm{MARGIN}>0$, $\mathrm{joint}(\mu^2,2\!\times\!10^{-3})>1$,
branch correct; the band is widened from $[\times0.5,\times2]$ to $(\times0,\times6.85)$, its
ceiling a physical branch boundary ($\mathrm{disc}=0$, $M_+$ ceases to exist), not a proof gap.

\section*{4. Monotonicity: analytic + interval-certified (operator review 2026-06-10)}

The window claims rest on three monotonicity facts, here promoted from a 3-point grid to
analytic + 400-point interval certificates (res5\_032 v1.1).

\noindent\textbf{(M1) $\mathrm{MARGIN}(\mu^2)$ strictly increasing.} $\mathrm{DIP\_BAND}=
|\widehat r_0|^{3/2}/(3\sqrt V)$ with $\widehat r_0=\mu^2-3U^2/(20V)<0$ on the range, so
$|\widehat r_0|=3U^2/(20V)-\mu^2$ is \emph{analytically} strictly decreasing $\Rightarrow
\mathrm{DIP\_BAND}$ strictly decreasing; $\mathrm{PB}(M_+)$ is interval-certified strictly
increasing (400-pt grid, min slope $d\,\mathrm{PB}/d\mu^2$ giving
$d\,\mathrm{MARGIN}/d\mu^2\ge0.095>0$). Hence $\mathrm{MARGIN}=\mathrm{PB}-\mathrm{DIP\_BAND}$
strictly increases (both terms favourable).

\noindent\textbf{(M2) $\mathrm{joint}$ strictly decreasing in $I$ (analytic).} With
$\rho=\mathrm{MARGIN}/C_2$ and $\mathrm{composed}=\mathrm{MARGIN}(1-1/\rho)=\mathrm{MARGIN}-C_2$,
\[
D(C_2)=(1+R_{\max})C_2+\frac{\mathrm{MARGIN}-C_2}{S_{\rm un}}
=C_2\Bigl(1+R_{\max}-\tfrac{1}{S_{\rm un}}\Bigr)+\frac{\mathrm{MARGIN}}{S_{\rm un}},\quad
1+R_{\max}-\tfrac{1}{S_{\rm un}}=0.500>0,
\]
so $D$ is strictly increasing in $C_2$; and $C_2=(\lambda I)^2 J_{\rm eff}\,n_{\rm pack}/(4(1-a_0))$
is strictly increasing in $I$ (the $I^2$ factor; interval-certified). Therefore
$\mathrm{joint}=\mathrm{MARGIN}/D$ is strictly decreasing in $I$, and $I_c^{\rm sel}$ is the
unique crossing $\mathrm{joint}=1$.

\noindent\textbf{(M3) $I_c^{\rm sel}(\mu^2)$ strictly increasing.} $\mathrm{joint}$ is
interval-certified strictly increasing in $\mu^2$; with (M2) the implicit function theorem gives
$dI_c/d\mu^2=-(\partial_{\mu^2}\mathrm{joint})/(\partial_I\mathrm{joint})>0$ (numerator $>0$,
denominator $<0$); verified on a dense $\mu^2$ grid. Hence the binding $I_c$ sits at the
low-$\mu^2$ edge, as used in Sec.~2.

\section*{5. The certified operating region}

\[
\mathcal R=\bigl\{(I,\mu^2):\ \mu^2\in(0,\mu^2_{\max}),\ I\in(0,I_c^{\rm sel}(\mu^2))\bigr\},
\qquad I_c^{\rm sel}\ \text{strictly increasing (M3)},
\]
operating point $(2\times10^{-3},0.005)$ interior. The enacted selection is stable throughout
$\mathcal R$; its boundaries $I_c^{\rm sel}$ and $\mu^2_{\max}$ are physical.
